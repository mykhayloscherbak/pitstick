67 /**
68  * @brief Applies current brightness level to the led values
69  * @param in Input led data
70  * @param out Calculated led data
71  */
72 static void applyBrightness(const Led_t * const in, Led_t * const out)
73 {
74  const Div_t coefs[MAX_BRIGHNESS_LEVELS] = {{1,8},{1,4},{1,2},{1,1}};
75  uint16_t tmp;
76  const uint8_t brighness = getBrightness();
77 
78  tmp = in->R * coefs[brighness].mul;
79  out->R = tmp / coefs[brighness].div;
80 
81  tmp = in->G * coefs[brighness].mul;
82  out->G = tmp / coefs[brighness].div;
83 
84  tmp = in->B * coefs[brighness].mul;
85  out->B = tmp / coefs[brighness].div;
86 
87  tmp = in->W * coefs[brighness].mul;
88  out->W = tmp / coefs[brighness].div;
89 }

98 static void displayDigit(const Colors_t color,uint8_t digit,uint8_t column)
99 {
100  column = (column > 1) ? 1 : column;
101  const uint8_t maxDigit = ((NLEDS / 2) / 4) * 3 + (NLEDS / 2) % 4;
102  digit = (digit > maxDigit) ? maxDigit : digit;
103  int16_t pos = column * (NLEDS -1);
104  const int8_t inc = (column == 0) ? 1 : -1;
105  for (uint8_t i = 0; i < digit / 3; i++)
106  {
107  for (uint8_t k = 0; k < 3; k++)
108  {
109  applyBrightness(&colors[color],leds + pos);
110  pos+=inc;
111  }
112  leds[pos] = colors[BLACK];
113  pos+=inc;
114  }
115 
116  for (uint8_t i = 0; i < digit % 3; i++)
117  {
118  applyBrightness(&colors[color],leds + pos);
119  pos+=inc;
120  }
121  if (inc > 0)
122  {
123  for (;pos < NLEDS / 2;pos += inc)
124  {
125  leds[pos]=colors[BLACK];
126  }
127  }
128  else
129  {
130  for (;pos >= NLEDS / 2; pos += inc)
131  {
132  leds[pos]=colors[BLACK];
133  }
134  }
135 }

229 uint8_t blinkTwice(const uint8_t init,const Colors_t color,pPowerColorFunc_t const getPowerColor)
230 {
231  static BlinkTwice_t state = BLINKTWICE_BLINK;
232  uint8_t changed = 0;
233  static uint32_t timer;
234  static uint8_t blinkCounter;
235  static uint8_t on;
236  if (init != 0)
237  {
238  ResetTimer(&timer);
239  blinkCounter = 4;
240  state = BLINKTWICE_BLINK;
241  showFull(color);
242  changed = !0;
243  }
244  switch (state)
245  {
246 
247  case BLINKTWICE_BLINK:
248 
249  if (IsExpiredTimer(&timer,500) != 0)
250  {
251  ResetTimer(&timer);
252  if (--blinkCounter > 0)
253  {
254  showFull(((blinkCounter & 1) != 0) ? BLACK: color);
255  }
256  else
257  {
258  state = BLINKTWICE_OFF;
259  showFull(BLACK);
260  ResetTimer(&timer);
261  on = 0;
262  }
263  changed = !0;
264  }
265  break;
266  case BLINKTWICE_OFF:
267  if (IsExpiredTimer(&timer, 500) != 0)
268  {
269  ResetTimer(&timer);
270  on = !on;
271  changed = !0;
272  showFull(BLACK);
273  if (on != 0 && getPowerColor != NULL)
274  {
275  put2pixels(getPowerColor(),0);
276  }
277  }
278  break;
279  default:
280  break;
281  }
282  return changed;
283 }
