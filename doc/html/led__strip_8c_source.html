70/**
71 * @brief Applies current brightness level to the led values
72 * @param in Input led data
73 * @param out Calculated led data
74 */
75static void applyBrightness(const Led_t * const in, Led_t * const out)
76{
77 const Div_t coefs[MAX_BRIGHNESS_LEVELS] = {{1,8},{1,4},{1,2},{1,1}};
78 uint16_t tmp;
79 const uint8_t brighness = getBrightness();
80
81 tmp = in->R * coefs[brighness].mul;
82 out->R = tmp / coefs[brighness].div;
83
84 tmp = in->G * coefs[brighness].mul;
85 out->G = tmp / coefs[brighness].div;
86
87 tmp = in->B * coefs[brighness].mul;
88 out->B = tmp / coefs[brighness].div;
89
90 tmp = in->W * coefs[brighness].mul;
91 out->W = tmp / coefs[brighness].div;
92}

101static void displayDigit(const Colors_t color,uint8_t digit,uint8_t column)
102{
103 column = (column > 1) ? 1 : column;
104 const uint8_t maxDigit = ((NLEDS / 2) / 4) * 3 + (NLEDS / 2) % 4;
105 digit = (digit > maxDigit) ? maxDigit : digit;
106 int16_t pos = column * (NLEDS -1);
107 const int8_t inc = (column == 0) ? 1 : -1;
108 for (uint8_t i = 0; i < digit / 3; i++)
109 {
110 for (uint8_t k = 0; k < 3; k++)
111 {
112 applyBrightness(&colors[color],leds + pos);
113 pos+=inc;
114 }
115 leds[pos] = colors[BLACK];
116 pos+=inc;
117 }
118
119 for (uint8_t i = 0; i < digit % 3; i++)
120 {
121 applyBrightness(&colors[color],leds + pos);
122 pos+=inc;
123 }
124 if (inc > 0)
125 {
126 for (;pos < NLEDS / 2;pos += inc)
127 {
128 leds[pos]=colors[BLACK];
129 }
130 }
131 else
132 {
133 for (;pos >= NLEDS / 2; pos += inc)
134 {
135 leds[pos]=colors[BLACK];
136 }
137 }
138}

267uint8_t blinkTwice(const uint8_t init,const Colors_t color,pPowerColorFunc_t const getPowerColor)
268{
269 static BlinkTwice_t state = BLINKTWICE_BLINK;
270 uint8_t changed = 0;
271 static uint32_t timer;
272 static uint8_t blinkCounter;
273 static uint8_t on;
274 if (init != 0)
275 {
276 ResetTimer(&timer);
277 blinkCounter = 4;
278 state = BLINKTWICE_BLINK;
279 showFull(color);
280 changed = !0;
281 }
282 switch (state)
283 {
284
285 case BLINKTWICE_BLINK:
286
287 if (IsExpiredTimer(&timer,500) != 0)
288 {
289 ResetTimer(&timer);
290 if (--blinkCounter > 0)
291 {
292 showFull(((blinkCounter & 1) != 0) ? BLACK: color);
293 }
294 else
295 {
296 state = BLINKTWICE_OFF;
297 showFull(BLACK);
298 ResetTimer(&timer);
299 on = 0;
300 }
301 changed = !0;
302 }
303 break;
304 case BLINKTWICE_OFF:
305 if (IsExpiredTimer(&timer, 500) != 0)
306 {
307 ResetTimer(&timer);
308 on = !on;
309 changed = !0;
310 showFull(BLACK);
311 if (on != 0 && getPowerColor != NULL)
312 {
313 put2pixels(getPowerColor(),0);
314 }
315 }
316 break;
317 default:
318 break;
319 }
320 return changed;
321}

324uint8_t blink(const uint8_t init, const Blink_t * const _blink)
325{
326 static uint8_t onPhase = 0;
327 static uint32_t timer = 0;
328 uint8_t changed = 0;
329 if ( 0 != init )
330 {
331 onPhase = !0;
332 ResetTimer(&timer);
333 changed = _blink->pOnPhase(!0);
334 }
335 else
336 {
337 if (0 != onPhase)
338 {
339 changed = _blink->pOnPhase(0);
340 if (IsExpiredTimer(&timer,_blink->on) != 0)
341 {
342 ResetTimer(&timer);
343 onPhase = 0;
344 changed = _blink->pOffPhase(!0);
345 }
346 }
347 else
348 {
349 changed = _blink->pOffPhase(0);
350 if (IsExpiredTimer(&timer,_blink->off) != 0)
351 {
352 ResetTimer(&timer);
353 onPhase = !0;
354 changed = _blink->pOnPhase(!0);
355 }
356 }
357 }
358 return changed;
359}
