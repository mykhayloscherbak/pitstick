20 GPIO_TypeDef *portOut;

23 if ( pinOut < 8 )
24 {

26 portOut->CRL |= (0b10u<< (pinOut * 4 + 2)); /* 0b10 Af PP */
27 portOut->CRL &= ~((0b01u<< (pinOut * 4 + 2))); /* 0b10 Af PP */
28 }
29 else
30 {
31 pinOut -= 8u;
32 portOut->CRH |= (0b11u<< (pinOut * 4)); /* 0b11 = 50Mhz */
33 portOut->CRH |= (0b10u<< (pinOut* 4 + 2)); /* 0b10 Af PP */
34 portOut->CRH &= ~((0b01u<< (pinOut * 4 + 2))); /* 0b10 Af PP */
35 }
36}
