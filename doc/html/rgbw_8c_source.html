26/**

30static void ConvertLeds(Led_t * const Leds)
31{
32 uint8_t i,k;
33
34 for (i=0; i < RESET_BITS; i++)
35 {
36 BitData[i] = 0;
37 }
38 BitData[sizeof(BitData) - 1] = 0;
39 for (i=0; i < NLEDS; i++)
40 {
41 Led_t CurrLed;
42 CurrLed.R = Leds[i].R;
43 CurrLed.G = Leds[i].G;
44 CurrLed.B = Leds[i].B;
45 CurrLed.W = Leds[i].W;
46 for ( k=0; k < 8; k++)
47 {
48 if ((CurrLed.G & (1 << k)) != 0 )
49 {
50 BitData[RESET_BITS + i * 32 + 7 - k] = CCR_1;
51 }
52 else
53 {
54 BitData[RESET_BITS + i * 32 + 7 - k] = CCR_0;
55 }
56 if ((CurrLed.R & (1 << k)) != 0 )
57 {
58 BitData[RESET_BITS + i * 32 + 8 + 7 - k] = CCR_1;
59 }
60 else
61 {
62 BitData[RESET_BITS + i * 32 + 8 + 7 - k] = CCR_0;
63 }
64 if ((CurrLed.B & (1 << k)) != 0 )
65 {
66 BitData[RESET_BITS + i * 32 + 16 + 7 - k] = CCR_1;
67 }
68 else
