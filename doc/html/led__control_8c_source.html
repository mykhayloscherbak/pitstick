299/**
300 * @brief show the param value and changes it during setup
301 * @param dV value and limits
302 * @param init This param is zero if it's not first call
303 * @return 0 if sending buffer to strip is not needed
304 */
305
306static uint8_t changeParam(DispValue_t * const dV,const uint8_t init)
307{
308 static States_t state = STATE_CONFIG_PARAM_IDLE;
309 static uint8_t saved = !0;
310 const uint8_t isPressed = IsPressed(B_CONFIG);
311 const uint8_t isLongPressed = IsLongPressed(B_CONFIG);
312 static uint32_t timer;
313 static uint32_t fastIncTimer;
314 uint8_t changed = 0;
315 if (init != 0)
316 {
317 state = STATE_CONFIG_PARAM_IDLE;
318 saved = !0;
319 if (dV->value < dV->min)
320 {
321 dV->value = dV->min;
322 saved = 0;
323 }
324 if (dV->value > dV->max)
325 {
326 dV->value = dV->max;
327 saved = 0;
328 }
329 dV->endMode = EM_CONTINUE;
330 }
331
332 switch(state)
333 {
334 case STATE_CONFIG_PARAM_IDLE:
335 if (isPressed)
336 {
337 state = STATE_CONFIG_PARAM_PRESSED;
338 }
339 else
340 {
341 state = STATE_CONFIG_PARAM_RELEASED;
342 ResetTimer(&timer);
343 }
344 dV->dispFunc(dV->value);
345 changed = !0;
346 break;
347 case STATE_CONFIG_PARAM_PRESSED:
348 if (isPressed == 0)
349 {
350 state = STATE_CONFIG_PARAM_RELEASED;
351 ResetTimer(&timer);
352 }
353 if (isLongPressed != 0)
354 {
355 ResetTimer(&fastIncTimer);
356 state = STATE_CONFIG_PARAM_LONG_PRESSED;
357 }
358 break;
359 case STATE_CONFIG_PARAM_RELEASED:
360 if (IsExpiredTimer(&timer,10000) != 0)
361 {
362 if (saved != 0)
363 {
364 state = STATE_CONFIG_PARAM_NOT_SAVING;
365 }
366 else
367 {
368 state = STATE_CONFIG_PARAM_SAVING;
369 }
370 } else if (isPressed != 0)
371 {
372 incValue(dV);
373 saved = 0;
374 state = STATE_CONFIG_PARAM_PRESSED;
375 dV->dispFunc(dV->value);
376 changed = !0;
377 }
378 break;
379 case STATE_CONFIG_PARAM_LONG_PRESSED:
380 if (isPressed == 0)
381 {
382 state = STATE_CONFIG_PARAM_RELEASED;
383 ResetTimer(&timer);
384 }
385 else if (IsExpiredTimer(&fastIncTimer,200) != 0)
386 {
387 ResetTimer(&fastIncTimer);
388 incValue(dV);
389 saved = 0;
390 changed = !0;
391 dV->dispFunc(dV->value);
392 }
393 break;
394 case STATE_CONFIG_PARAM_SAVING:
395 showFull(BLACK);
396 for (uint8_t i = 0; i < 20; i++)
397 {
398 put2pixels(DARK_RED,i);
399 }
400 changed = !0;
401 ResetTimer(&timer);
402 state = STATE_CONFIG_PARAM_WAITING;
403 dV->endMode = EM_CONTINUE_DO_NOT_UPDATE;
404 break;
405 case STATE_CONFIG_PARAM_NOT_SAVING:
406 showFull(BLACK);
407 changed = !0;
408 ResetTimer(&timer);
409 state = STATE_CONFIG_PARAM_WAITING;
410 dV->endMode = EM_CONTINUE_DO_NOT_UPDATE;
411 break;
412 case STATE_CONFIG_PARAM_WAITING:
413 if (IsExpiredTimer(&timer,1000) != 0)
414 {
415 dV->endMode = (saved != 0) ? EM_END_NOT_SAVING : EM_END_SAVING;
416 }
417 else
418 {
419 dV->endMode = EM_CONTINUE_DO_NOT_UPDATE;
420 }
421 break;
422 default:
423 break;
424 }
425 return changed;
426}

428/**
429 * @brief all config mode processing is done here
430 * @param nextState pointer to the result. Non zero value shows that configuration is done and you can pass
431 * to the next step
432 * @return nonzero value shows that data is updated and must be sent to the strip
433 */
434static uint8_t config(uint8_t * const nextState)
435{
436 uint8_t * conf = eeemuGetValue();
437 static uint8_t brightness;
438 static States_t state = STATE_CONFIG_BEGIN;
439 static uint8_t saved = !0;
440 static uint32_t timer;
441 uint8_t changed = 0;
442 static uint8_t savedParams[MAX_STORED];
443 static DispValue_t dV;
444 if (nextState != NULL)
445 {
446 *nextState = 0;
447 }
448
449 switch(state)
450 {
451 case STATE_CONFIG_BEGIN:
452 memcpy(savedParams,conf,MAX_STORED);
453 brightness = savedParams[CH_BRIGHTNESS];
454 dV.dispFunc = displayBrightness;
455 dV.min = 0;
456 dV.max = MAX_BRIGHNESS_LEVELS - 1;
457 dV.value = brightness;
458 dV.endMode = EM_CONTINUE;
459 changed = changeParam(&dV,!0);
460 brightness = dV.value;
461 state = STATE_CONFIG_BRIGHTNESS;
462 break;
463 case STATE_CONFIG_BRIGHTNESS:
464 changed = changeParam(&dV,0);
465 brightness = dV.value;
466 setBrightness(brightness);
467 savedParams[CH_BRIGHTNESS] = brightness;
468 if (dV.endMode != EM_CONTINUE && dV.endMode != EM_CONTINUE_DO_NOT_UPDATE)
469 {
470 if (dV.endMode == EM_END_SAVING)
471 {
472 saved = 0;
473 }
474 savedParams[CH_BRIGHTNESS] = brightness;
475 dV.dispFunc = dispModePattern;
476 dV.min = 0;
477 dV.max = MODE_TOTAL - 1;
478 dV.value = savedParams[CH_MODE];
479 dV.endMode = EM_CONTINUE;
480 uint8_t changed2 = changeParam(&dV,!0);
481 changed = changed || changed2;
482 state = STATE_CONFIG_MODE;
483 }
484 else
485 {
486 if (changed != 0 && dV.endMode != EM_CONTINUE_DO_NOT_UPDATE)
487 {
488 dispBrighnessPattern();
489 }
490 }
491 break;
492 case STATE_CONFIG_MODE:
493 changed = changeParam(&dV,0);
494 if (dV.endMode != EM_CONTINUE && dV.endMode != EM_CONTINUE_DO_NOT_UPDATE)
495 {
496 if (dV.endMode == EM_END_SAVING)
497 {
498 saved = 0;
499 }
500 savedParams[CH_MODE] = dV.value;
501 uint8_t noParam = !0;

530 break;

533 noParam = !0;
534 break;
535 case MODE_KART2H:
536 state = STATE_CONFIG_K2H_END;
537 noParam = !0;
538 break;
539 case MODE_PITINVITE:

556 break;
557 }
558 if (noParam == 0)
559 {
560 const uint8_t changed2 = changeParam(&dV,!0);

562 }

565 changed = !0;

1077 */
1078typedef struct
1079{

1084/**
1085 * @brief phase function processor. Calls needed phase function every time this function is called starting with
1086 * @brief @ref Phase_desc_t::start field of it's descriptor.
1087 * @param desc Descriptor table
1088 * @param nextState out parameter. Becomes nonzero when all table is processed
1089 * @param ms millisecons from the start
1090 * @return nonzero means the led strip must be updated
1091 */
1092static uint8_t processPhaseTable(const Phase_desc_t * const desc,uint8_t * const nextState,const uint32_t ms)
1093{
1094 if (nextState != NULL)
1095 {
1096 *nextState = 0;
1097 }
1098 static uint8_t oldPos = 99;
1099 uint8_t i = 0;
1100 for (; desc[i].pPhase != NULL;i++)
1101 {
1102 if (ms >= desc[i].start && ms < desc[i + 1].start)
1103 {
1104 break;
1105 }
1106 }
1107 uint8_t changed = 0;
1108 if (desc[i].pPhase == NULL)
1109 {
1110 if (nextState != NULL)
1111 {
1112 *nextState = !0;
1113 oldPos = 99;
1114 }
1115 }
1116 else
1117 {
1118 changed = desc[i].pPhase(i != oldPos);
1119 oldPos = i;
1120 }
1121 return changed;
1122}
