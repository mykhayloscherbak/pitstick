243 /**
244  * @brief show the param value and changes it during setup
245  * @param dV value and limits

247  * @return 0 if sending buffer to strip is not needed

250 static uint8_t changeParam(DispValue_t * const dV,const uint8_t init)
251 {
252  static States_t state = STATE_CONFIG_PARAM_IDLE;
253  static uint8_t saved = !0;
254  const uint8_t isPressed = IsPressed(B_CONFIG);
255  const uint8_t isLongPressed = IsLongPressed(B_CONFIG);
256  static uint32_t timer;
257  static uint32_t fastIncTimer;
258  uint8_t changed = 0;
259  if (init != 0)
260  {
261  state = STATE_CONFIG_PARAM_IDLE;
262  saved = !0;
263  if (dV->value < dV->min)
264  {
265  dV->value = dV->min;
266  saved = 0;
267  }
268  if (dV->value > dV->max)
269  {
270  dV->value = dV->max;
271  saved = 0;
272  }
273  dV->endMode = EM_CONTINUE;
274  }
275 
276  switch(state)
277  {
278  case STATE_CONFIG_PARAM_IDLE:
279  if (isPressed)
280  {
281  state = STATE_CONFIG_PARAM_PRESSED;
282  }
283  else
284  {
285  state = STATE_CONFIG_PARAM_RELEASED;
286  ResetTimer(&timer);
287  }
288  dV->dispFunc(dV->value);
289  changed = !0;
290  break;
291  case STATE_CONFIG_PARAM_PRESSED:
292  if (isPressed == 0)
293  {
294  state = STATE_CONFIG_PARAM_RELEASED;
295  ResetTimer(&timer);
296  }
297  if (isLongPressed != 0)
298  {
299  ResetTimer(&fastIncTimer);
300  state = STATE_CONFIG_PARAM_LONG_PRESSED;
301  }
302  break;
303  case STATE_CONFIG_PARAM_RELEASED:
304  if (IsExpiredTimer(&timer,10000) != 0)
305  {
306  if (saved != 0)
307  {
308  state = STATE_CONFIG_PARAM_NOT_SAVING;
309  }
310  else
311  {
312  state = STATE_CONFIG_PARAM_SAVING;
313  }
314  } else if (isPressed != 0)
315  {
316  incValue(dV);
317  saved = 0;
318  state = STATE_CONFIG_PARAM_PRESSED;
319  dV->dispFunc(dV->value);
320  changed = !0;
321  }
322  break;
323  case STATE_CONFIG_PARAM_LONG_PRESSED:
324  if (isPressed == 0)
325  {
326  state = STATE_CONFIG_PARAM_RELEASED;
327  ResetTimer(&timer);
328  }
329  else if (IsExpiredTimer(&fastIncTimer,200) != 0)
330  {
331  ResetTimer(&fastIncTimer);
332  incValue(dV);
333  saved = 0;
334  changed = !0;
335  dV->dispFunc(dV->value);
336  }
337  break;
338  case STATE_CONFIG_PARAM_SAVING:
339  showFull(BLACK);
340  for (uint8_t i = 0; i < 20; i++)
341  {
342  put2pixels(DARK_RED,i);
343  }
344  changed = !0;
345  ResetTimer(&timer);
346  state = STATE_CONFIG_PARAM_WAITING;
347  dV->endMode = EM_CONTINUE_DO_NOT_UPDATE;
348  break;
349  case STATE_CONFIG_PARAM_NOT_SAVING:
350  showFull(BLACK);
351  changed = !0;
352  ResetTimer(&timer);
353  state = STATE_CONFIG_PARAM_WAITING;
354  dV->endMode = EM_CONTINUE_DO_NOT_UPDATE;
355  break;
356  case STATE_CONFIG_PARAM_WAITING:
357  if (IsExpiredTimer(&timer,1000) != 0)
358  {
359  dV->endMode = (saved != 0) ? EM_END_NOT_SAVING : EM_END_SAVING;
360  }
361  else
362  {
363  dV->endMode = EM_CONTINUE_DO_NOT_UPDATE;
364  }
365  break;
366  default:
367  break;
368  }
369  return changed;
370 }

372 /**
373  * @brief all config mode processing is done here

388  if (nextState != NULL)
389  {
390  *nextState = 0;
391  }
392 
393  switch(state)
394  {
395  case STATE_CONFIG_BEGIN:
396  memcpy(savedParams,conf,MAX_STORED);
397  brightness = savedParams[CH_BRIGHTNESS];
398  dV.dispFunc = displayBrightness;
399  dV.min = 0;
400  dV.max = MAX_BRIGHNESS_LEVELS - 1;
401  dV.value = brightness;
402  dV.endMode = EM_CONTINUE;
403  changed = changeParam(&dV,!0);
404  brightness = dV.value;

408  changed = changeParam(&dV,0);

425  changed = changed || changed2;

674 /**

676  * @return Color index
677  */
678 static Colors_t getPowerColor(void)
679 {
680  static const V2Color_t V2Color[]=
681  {
682  {6300,REDDER},
683  {6500,ORANGE},
684  {7000,YELLOW},
685  {7600,GREEN}
686  };
687  const uint16_t mv = GetAdc_Voltage();
688  Colors_t retval;
689  if (mv <= V2Color[0].mv)
690  {
691  retval = RED;
692  }
693  else
694  {
695  uint8_t i;
696  const uint8_t arrsize = sizeof(V2Color)/sizeof(V2Color[0]);
697  for (i = 0; i < arrsize; i++)

910 /**
911  * @brief Phase description
912  */
913 typedef struct
914 {
915  const uint32_t start; /**< Phase start time. ms from turning the stick on */

919 /**
920  * @brief phase function processor. Calls needed phase function every time this function is called starting with
921  * @brief @ref Phase_desc_t::start field of it's descriptor.
922  * @param desc Descriptor table
923  * @param nextState out parameter. Becomes nonzero when all table is processed
924  * @param ms millisecons from the start
925  * @return nonzero means the led strip must be updated
926  */
927 static uint8_t processPhaseTable(const Phase_desc_t * const desc,uint8_t * const nextState,const uint32_t ms)
928 {
929  if (nextState != NULL)
930  {
931  *nextState = 0;
932  }
933  static uint8_t oldPos = 99;
934  uint8_t i = 0;
935  for (; desc[i].pPhase != NULL;i++)
936  {
937  if (ms >= desc[i].start && ms < desc[i + 1].start)
938  {
939  break;
940  }
941  }
942  uint8_t changed = 0;
943  if (desc[i].pPhase == NULL)
944  {
945  if (nextState != NULL)
946  {
947  *nextState = !0;
948  oldPos = 99;
949  }
950  }
951  else
952  {
953  changed = desc[i].pPhase(i != oldPos);
954  oldPos = i;
955  }
956  return changed;
957 }

967  * @param[in] ms millisecons from the start

1381  }
1382  }
1383  break;
