40void Gpio_Init(void)
41{
42 Gpio_Desc_t Pin;
43 RCC->APB2ENR |= RCC_APB2ENR_IOPAEN | RCC_APB2ENR_IOPCEN | RCC_APB2ENR_AFIOEN;
44 AFIO->MAPR |= AFIO_MAPR_SWJ_CFG_1;
45 AFIO->MAPR &= ~(AFIO_MAPR_SWJ_CFG_0 | AFIO_MAPR_SWJ_CFG_2);
46 for ( Pin = 0; Pin < GPIO_TOTAL; Pin++)
47 {
48 if ( Gpio_Config[Pin].Pin < 8 )
49 {
50 if ( Gpio_Config[Pin].Mode == GPIO_MODE_IN)
51 {
52 (Gpio_Config[Pin].Port)->CRL &= ~( 0b11u<< ( Gpio_Config[Pin].Pin * 4));
53 (Gpio_Config[Pin].Port)->CRL &= ~( 0b01u<< ( Gpio_Config[Pin].Pin * 4 + 2));
54 (Gpio_Config[Pin].Port)->CRL |= ( 0b10u<< ( Gpio_Config[Pin].Pin * 4 + 2));
55 (Gpio_Config[Pin].Port)->ODR |= ( 1u<< Gpio_Config[Pin].Pin ); /* Pull-Up */
56 }
57 else
58 {
59 if ( Gpio_Config[Pin].Mode == GPIO_MODE_OUT )
60 {
61 (Gpio_Config[Pin].Port)->CRL &= ~( 0b10u<< ( Gpio_Config[Pin].Pin * 4));
62 (Gpio_Config[Pin].Port)->CRL |= ( 0b10u<< ( Gpio_Config[Pin].Pin * 4));
63 (Gpio_Config[Pin].Port)->CRL &= ~( 0b11u<< ( Gpio_Config[Pin].Pin * 4 + 2));
64 (Gpio_Config[Pin].Port)->BRR |= ( 1u<< Gpio_Config[Pin].Pin );
65 }
66 }
67 }
68 else
69 {
70 if ( Gpio_Config[Pin].Mode == GPIO_MODE_IN)
71 {
72 (Gpio_Config[Pin].Port)->CRH &= ~( 0b11u<< ( (Gpio_Config[Pin].Pin - 8) * 4));
73 (Gpio_Config[Pin].Port)->CRH &= ~( 0b01u<< ( (Gpio_Config[Pin].Pin - 8) * 4 + 2));
74 (Gpio_Config[Pin].Port)->CRH |= ( 0b10u<< ( (Gpio_Config[Pin].Pin - 8) * 4 + 2));
75 (Gpio_Config[Pin].Port)->ODR |= ( 1u<< Gpio_Config[Pin].Pin ); /* Pull-Up */
76 }
77 else
78 {
79 if ( Gpio_Config[Pin].Mode == GPIO_MODE_OUT )
80 {
81 (Gpio_Config[Pin].Port)->CRH &= ~( 0b10u<< ( (Gpio_Config[Pin].Pin - 8) * 4));
82 (Gpio_Config[Pin].Port)->CRH |= ( 0b10u<< ( (Gpio_Config[Pin].Pin - 8) * 4));
83 (Gpio_Config[Pin].Port)->CRH &= ~( 0b11u<< ( (Gpio_Config[Pin].Pin -8 ) * 4 + 2));
84 (Gpio_Config[Pin].Port)->BRR |= ( 1u<< Gpio_Config[Pin].Pin );
85 }
86 }
87 }
88 }
