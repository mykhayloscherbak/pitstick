37 /* Channels 0 sample rate is 239.5 cycles */
38 ADC1->SMPR2=ADC_SMPR2_SMP0_0|ADC_SMPR2_SMP0_1|ADC_SMPR2_SMP0_2;

44

51 DMA1_Channel1->CMAR=(uint32_t)Adc_Buf;
52 DMA1_Channel1->CNDTR=sizeof(Adc_Buf)/sizeof(uint16_t);
53 /* RG p. 277 */

55 DMA_CCR_MSIZE_0| /* Memory and per. size = 16 bit */
56 DMA_CCR_PSIZE_0|

58 DMA1_Channel1->CCR |= DMA_CCR_TCIE;
59

69/**
70 * @brief DMA ISR and "entropy" calcualations. Entropy is number of different adc readings for Vbat
71 */
72void DMA1_Channel1_IRQHandler(void)
73{
74 if ((DMA1->ISR & DMA_ISR_TCIF1) != 0)
75 {
76 DMA1->IFCR = DMA_IFCR_CTCIF1;
77 uint16_t tmp = Adc_Buf[0];
78 for (uint8_t i = 1; i < ADC_AVG; i++)
79 {
80 if (tmp != Adc_Buf[i])
81 {
82 entropy++;
83 }
84 tmp = Adc_Buf[i];
85 }
86 }
87}
