66/**
67 * @brief crc8 calcualation for data block
68 * @param data data itself
69 * @param len data length
70 * @return crc8
71 */
72static uint8_t crc8(uint8_t *data, uint8_t const len)
73{
74 uint8_t crc = 0xff;
75 uint8_t i, j;
76 for (i = 0; i < len; i++) {
77 crc ^= data[i];
78 for (j = 0; j < 8; j++) {
79 if ((crc & 0x80) != 0)
80 crc = (uint8_t)((crc << 1) ^ 0x31);
81 else
82 crc <<= 1;
83 }
84 }
85 return crc;
86}

92static void eeemuErasePage(uint32_t const addr)
93{
94 waitBusy();
95 FLASH->KEYR = 0x45670123;
96 FLASH->KEYR = 0xCDEF89AB;
97 FLASH->CR |= FLASH_CR_PER;
98 FLASH->AR = addr;
99 FLASH->CR |= FLASH_CR_STRT;
100 waitBusy();
101 FLASH->CR &= ~FLASH_CR_PER;
102 FLASH->CR |= FLASH_CR_LOCK;
103}

123void eeemu_write(uint8_t * const values)
124{
125 uint8_t i = 0;
126 for (i = 0 ; i < sizeof(eeemu_storage_s)/sizeof(uint16_t); i++)
127 {
128 if (eeemu_array[next_pos].raw[i] != 0xFFFF)
129 {
130 break;
131 }
132 }
133
134 if (i < sizeof(eeemu_storage_s)/sizeof(uint16_t) || next_pos == MAXSTORAGE) /* Wrong record or page full */
135 {
136 eeemuErasePage((uint32_t)(&eeemu_array[0]));
137 next_pos = 0;
138 }
139 eeemu_storage_t u;
140 memcpy(u.structured.stored,values,MAX_STORED);
141 u.structured.crc8 = crc8(values,MAX_STORED);
142 waitBusy();
143 FLASH->KEYR = 0x45670123;
144 FLASH->KEYR = 0xCDEF89AB;
145 FLASH->CR |= FLASH_CR_PG;
146 for (uint8_t i = 0; i < sizeof(eeemu_storage_t)/sizeof(uint16_t); i++ )
147 {
148 eeemu_array[next_pos].raw[i] = u.raw[i];
149 waitBusy();
150 }
151 FLASH->CR &= ~FLASH_CR_PG;
152 FLASH->CR |= FLASH_CR_LOCK;
153 next_pos++;
154
155}
