47void Button_Process(const Buttons_id_t Button_id)
48{
49 volatile Buttons_t *button=Buttons+Button_id;
50 if (button->state==NotExists) return;
51 if ( button->buttonFunc() == 0) /* Button is released */
52 switch (button->state)
53 {
54
55 case LongReleased:
56 return;
57 case SteadyReleased:
58 if ((button->count++)>=BUTTONLONGWAIT)
59 {
60 button->count=BUTTONLONGWAIT;
61 button->state=LongReleased;
62 }
63 return;
64 case Released:
65 if ((button->count++)>=BUTTONWAIT)
66 {
67 button->count=BUTTONWAIT;
68 button->state=SteadyReleased;
69 }
70 return;
71 default:
72 button->count=0;
73 button->state=Released;
74 return;
75 }
76 else
77 {
78 switch (button->state)
79 { /* Button is pressed */
80 case LongPressed:
81 return;
82 case SteadyPressed:
83 if ((button->count++)>=BUTTONLONGWAIT)
84 {
85 button->count=BUTTONLONGWAIT;
86 button->state=LongPressed;
87 }
88 return;
89 case Pressed:
90 if ((button->count++)>=BUTTONWAIT)
91 {
92 button->count=BUTTONWAIT;
93 button->state=SteadyPressed;
94 }
95 return;
96 default:
97 button->count=0;
98 button->state=Pressed;
99 return;
100 }
101 }
102}
